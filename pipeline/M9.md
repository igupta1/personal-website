## Module shapes

### 1. `api/generate-leads.js`

```javascript
const { list } = require('@vercel/blob');

const BLOB_KEY = 'leads-current.json';
const VALID_NICHES = ['it_msp', 'mssp', 'cloud'];
const CACHE_HEADER = 'public, s-maxage=300, stale-while-revalidate=3600';

let _cachedBlobUrl = null;

async function _resolveBlobUrl() {
  // The store URL contains a per-store hostname we don't know upfront;
  // list() returns the canonical public URL. Cached in module memory
  // because the URL is stable until the blob is deleted (overwrite
  // preserves the URL).
  if (_cachedBlobUrl) return _cachedBlobUrl;
  const { blobs } = await list({ prefix: BLOB_KEY, limit: 1 });
  if (!blobs.length) throw new Error('blob not found');
  _cachedBlobUrl = blobs[0].url;
  return _cachedBlobUrl;
}

module.exports = async function handler(req, res) {
  if (req.method !== 'GET') {
    res.setHeader('Allow', 'GET');
    return res.status(405).json({ error: 'method not allowed' });
  }

  const niche = req.query.niche;
  if (niche !== undefined && !VALID_NICHES.includes(niche)) {
    return res.status(400).json({
      error: `niche must be one of: ${VALID_NICHES.join(', ')}`,
    });
  }

  try {
    const blobUrl = await _resolveBlobUrl();
    const blobResp = await fetch(blobUrl, { cache: 'no-store' });
    if (!blobResp.ok) {
      throw new Error(`blob fetch failed: ${blobResp.status}`);
    }
    const payload = await blobResp.json();

    res.setHeader('Cache-Control', CACHE_HEADER);
    if (niche) {
      return res.status(200).json({
        generated_at: payload.generated_at,
        niche,
        leads: payload.niches[niche] || [],
      });
    }
    return res.status(200).json(payload);
  } catch (err) {
    console.error('generate-leads failed:', err);
    return res.status(502).json({ error: 'upstream blob fetch failed' });
  }
};
```

### 2. `src/pages/LeadsPage.js` — shared page component

```jsx
import React, { useEffect, useState, useMemo } from "react";
import LeadCard from "../components/LeadCard";
import LeadFilters from "../components/LeadFilters";

export default function LeadsPage({ niche, title, subtitle }) {
  const [leads, setLeads] = useState(null);   // null = loading
  const [error, setError] = useState(null);
  const [generatedAt, setGeneratedAt] = useState(null);

  const [industries, setIndustries] = useState(new Set());
  const [headcountMax, setHeadcountMax] = useState(250);
  const [search, setSearch] = useState("");
  const [sortBy, setSortBy] = useState("score"); // score | name

  useEffect(() => {
    document.title = `${title} — Ishaan Gupta`;
    fetch(`/api/generate-leads?niche=${niche}`)
      .then(r => r.ok ? r.json() : Promise.reject(new Error(`HTTP ${r.status}`)))
      .then(d => { setLeads(d.leads); setGeneratedAt(d.generated_at); })
      .catch(e => setError(e.message));
  }, [niche, title]);

  const visibleLeads = useMemo(() => {
    if (!leads) return [];
    let out = leads;
    if (industries.size > 0) {
      out = out.filter(l => l.industry && industries.has(l.industry));
    }
    if (headcountMax < 250) {
      out = out.filter(l => l.headcount === null || l.headcount <= headcountMax);
    }
    if (search.trim()) {
      const q = search.trim().toLowerCase();
      out = out.filter(l => l.name.toLowerCase().includes(q));
    }
    if (sortBy === "name") {
      out = [...out].sort((a, b) => a.name.localeCompare(b.name));
    }
    // score sort: server already returns score DESC nulls last; use as-is
    return out;
  }, [leads, industries, headcountMax, search, sortBy]);

  if (error) return <ErrorState message={error} />;
  if (leads === null) return <LoadingState />;

  return (
    <div className="leads-page">
      <header className="leads-header">
        <h1>{title}</h1>
        {subtitle && <p className="subtitle">{subtitle}</p>}
        {generatedAt && <p className="meta">Updated {formatRelative(generatedAt)}</p>}
      </header>
      <LeadFilters
        leads={leads}
        industries={industries} setIndustries={setIndustries}
        headcountMax={headcountMax} setHeadcountMax={setHeadcountMax}
        search={search} setSearch={setSearch}
        sortBy={sortBy} setSortBy={setSortBy}
      />
      <div className="leads-grid">
        {visibleLeads.map((lead, i) => (
          <LeadCard key={`${lead.name}-${i}`} lead={lead} />
        ))}
        {visibleLeads.length === 0 && (
          <div className="empty">No leads match the current filters.</div>
        )}
      </div>
    </div>
  );
}
```

```jsx
import React, { useState } from "react";

export default function LeadCard({ lead }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <div className="lead-card">
      <div className="lead-card-header">
        <div>
          <h3 className="lead-name">{lead.name}</h3>
          <div className="lead-meta">
            {lead.industry && <span className="chip">{prettyIndustry(lead.industry)}</span>}
            {lead.headcount && <span className="chip">~{lead.headcount} emp</span>}
            {(lead.city || lead.state) && (
              <span className="chip">
                {[lead.city, lead.state].filter(Boolean).join(", ")}
              </span>
            )}
          </div>
        </div>
        <ScoreBadge score={lead.score} />
      </div>

      {lead.insight && <p className="insight">{lead.insight}</p>}

      <button
        className="expand-toggle"
        onClick={() => setExpanded(e => !e)}
      >
        {expanded ? "Hide outreach ↑" : "Show outreach ↓"}
      </button>

      {expanded && (
        <div className="lead-card-details">
          <div className="outreach-block">
            <div className="outreach-actions">
              <button onClick={() => copyToClipboard(lead.outreach)}>
                Copy outreach
              </button>
            </div>
            <pre className="outreach-text">{lead.outreach || "(no outreach copy)"}</pre>
          </div>

          <details className="signals-block">
            <summary>Signals ({lead.signals.length})</summary>
            <ul>
              {lead.signals.map((s, i) => (
                <li key={i}>
                  <strong>{s.type}</strong> · {s.days_ago}d ago
                  {s.payload && Object.keys(s.payload).length > 0 && (
                    <pre className="signal-payload">{JSON.stringify(s.payload, null, 2)}</pre>
                  )}
                </li>
              ))}
            </ul>
          </details>
        </div>
      )}
    </div>
  );
}
```

## Locked-pending-redline decisions

1. **Styling — Tailwind for net-new components.** `CLAUDE.md` says
   "follow existing patterns — Tailwind." Existing pages don't actually
   use Tailwind, but the dependency is installed and configured. M9
   uses Tailwind for `LeadCard`, `LeadFilters`, `LeadsPage`. Layout
   chrome (page padding, max-width) follows the existing `.about-page-combined`
   pattern via a small wrapper for visual consistency.
2. **API behavior — return everything, no thresholding.** Per M7
   redline "show everything; UI sorts." The function forwards the blob
   verbatim (or sliced by niche). UI does sort/filter/search.
3. **API caching — `Cache-Control: public, s-maxage=300, stale-while-revalidate=3600`.**
   Vercel edge cache holds responses for 5 min. With M10's nightly
   cron updating once a day, this is more than adequate; almost no
   request hits the function.
4. **Blob URL discovery — `list({ prefix: 'leads-current.json' })`**
   in the function, cached in module-level `let _cachedBlobUrl`.
   Module memory survives across invocations on warm Vercel functions.
   Alternative: hardcode the blob URL in env (brittle if the store
   ever changes) or in code (worse).
5. **Header.js — DO NOT modify.** Per `CLAUDE.md` it's protected.
   Pages are reachable via direct URL (`/it-msps`, `/mssps`, `/cloud`)
   only. **OPEN QUESTION (needs your call):** without nav links the
   pages are undiscoverable from the homepage. Three options:
     - **(a)** Don't touch Header, accept undiscoverable (purist).
     - **(b)** Add three nav links to Header — modifies a "protected"
       file but is a minimal append. (My recommendation.)
     - **(c)** Add a "Lead generators" section to AITools.js (`/gtm`)
       linking to the three pages. Modifies AITools.js (also protected).
6. **App.js — add three new routes only.** The four existing lines
   (BrowserRouter, Header, Footer, Navigate-to-/) stay untouched. New
   routes go above the `<Route path="*" />` redirect so the catch-all
   still works.
7. **LeadCard collapsed-by-default.** Outreach is hidden behind "Show
   outreach ↓" — keeps the grid scannable. Insight (one sentence) is
   always visible.
8. **Copy-to-clipboard via `navigator.clipboard.writeText()`** — modern
   browsers only. Fallback toast "Copied!" via simple `setTimeout`
   state. No external toast lib.
9. **Filter controls on LeadFilters.js — keep minimal in v1:**
   industry multi-select, headcount-max slider (1–250), name search,
   sort-by dropdown (`score` default, `name` alternative). No date
   ranges, no signal-type filter. Add later if the user finds them
   missing in real use.
10. **Empty/loading/error states.** All three handled inline. No
    skeleton loaders in v1 — a simple "Loading…" string.
11. **Score badge color** — green ≥70, amber 40–69, gray <40 / null.
    Provides at-a-glance quality without the user having to read the
    number. Tunable.
12. **Industry tag rendering** — convert enum value
    (`legal_professional`) to a display label (`Legal & Professional`)
    via a small mapping in `LeadCard.js`. Keeps the JSON schema
    canonical (enum values).
13. **No JS tests in M9.** The repo has no existing component tests
    (just CRA's default test runner which goes unused). Verification
    is manual browser smoke against `localhost:3000` (with `vercel
    dev` for the API) and against the deployed preview.
14. **Local dev — `vercel dev`.** Documented in pipeline/CLAUDE.md
    additions. `npm start` alone won't have the API; `vercel dev` runs
    both the React app and the serverless function locally.
15. **Commit granularity — single commit `add generate-leads api and
    react pages`** since the React + API are coupled. Touches root
    (api/, src/, App.js) but no `pipeline/` files except this plan
    doc.

## Out of scope (per `pipeline/ROADMAP.md`)

- Manual lead curation UI (mark-as-contacted, dismiss, snooze) — defer.
- Auth on the generate-leads endpoint — blob is public; consumers are
  the React UI only. Add auth when/if private listings are needed.
- Rate limiting on the API — Vercel edge cache + `s-maxage=300` makes
  this a non-issue at our traffic.
- Score history / trends in the UI — needs persisted score history;
  add when the data exists.
- A/B testing different outreach prompts in the UI — M6 out-of-scope.
- Analytics / tracking — defer.
- Pagination — at SMB-targeting volume the per-niche list is in the
  hundreds; render-all is fine. Add virtualization (react-window) if
  the UI gets sluggish.
- Mobile-specific layout polish beyond Tailwind responsive defaults.
